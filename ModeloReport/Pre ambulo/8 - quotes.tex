\newcommand{\longcite}[3]{%
\ifstrequal{#3}{pos}{}{#1}:

\begin{quoting}[rightmargin=0cm,leftmargin=\indentquote]
\footnotesize\linespread{1}\selectfont\noindent#2\ifstrequal{#3}{pos}{#1}{}.
\end{quoting}
}
%======================================================================
\newcommand{\shortcite}[3]{%
\ifstrequal{#3}{pos}{}{#1} ``#2"%
\ifstrequal{#3}{pos}{#1}{}%
}
